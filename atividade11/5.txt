#include <stdio.h>

#define PI 3.141592

float calcularVolumeEsfera(float raio) {
    return (4.0 / 3.0) * PI * raio * raio * raio;
}

int main() {
    float raio;
    printf("Digite o raio da esfera: ");
    scanf("%f", &raio);
    float volume = calcularVolumeEsfera(raio);
    printf("O volume da esfera Ã©: %.2f", volume);
    return 0;
}
